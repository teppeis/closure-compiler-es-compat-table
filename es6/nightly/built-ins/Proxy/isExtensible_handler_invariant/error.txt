  var passed = false;

  try {
    Object.isExtensible(new Proxy({}, {
      isExtensible: function (t) {
        passed = true;
        return false;
      }
    }));
    return false;
  } catch(e) {}
  try {
    Object.isExtensible(new Proxy(Object.preventExtensions({}), {
      isExtensible: function (t) {
        return true;
      }
    }));
    return false;
  } catch(e) {}
  return true;

};
